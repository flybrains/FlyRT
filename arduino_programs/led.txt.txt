int val=0;

void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
  Serial.begin(9600);
}

void loop() {

  while(Serial.available()==0)
  {
    digitalWrite(LED_BUILTIN, LOW);
  }

  if (Serial.available() > 0)
  {
    val = Serial.read();

    if (val == '1')
    {
       digitalWrite(LED_BUILTIN, HIGH);
    }

    if (val == '0')
    {
       digitalWrite(LED_BUILTIN, LOW);
    }
    
  }
  
}
